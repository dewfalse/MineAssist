# Permission description
# mineassist.assist: Disable assist when Main.Use Permissions is true.
# mineassist.global: Toggle global MineAssist ON/OFF.
# mineassist.reload: Reload config.
# mineassist.toggle: Toggle global/player MineAssist ON/OFF.
# mineassist.tools: Add/Remove tools.
# mineassist.ores: Add/Remove target ores.
#
# Permission examples
# ======================================
# groups:
#     default:
#         permissions:
#             mineassist.assist: false
#             mineassist.global: false
#             mineassist.reload: false
#             mineassist.toggle: false
#             mineassist.tools: false
#             mineassist.ores: false
#
